00605 {

00607     switch (selector){
00608         case 0: quantile = 2.326348; break;
00609         case 1: quantile = 2.053749; break;

00611     }
00612     double cutoff = 999999.0;
00613     for (size_t i = 0; i < mu.size(); ++i)
